Task 5
SELECT TOP 5 WITH TIES c.u_id, c.cnt
FROM
(
SELECT u.id AS u_id, COUNT (1) AS cnt
FROM order o JOIN user u ON o.user_id=u.id
JOIN product p ON o.product_id=p.id
GROUP BY u.id
) c
ORDER BY c.cnt DESC
